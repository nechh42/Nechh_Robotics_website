Örnek Python
import random
import numpy as np

trades = [1.56, 2.16, 1.60, 2.48, -0.30]

simulations = 1000
results = []

for _ in range(simulations):

    equity = 0
    peak = 0
    max_dd = 0

    shuffled = random.sample(trades, len(trades))

    for t in shuffled:

        equity += t
        peak = max(peak, equity)
        dd = peak - equity
        max_dd = max(max_dd, dd)

    results.append(max_dd)

print("Worst DD:", max(results))
print("Average DD:", np.mean(results))

🧪 Python Örneği
def strategy_score(stats):

    winrate = stats["winrate"]
    pf = stats["profit_factor"]
    expectancy = stats["expectancy"]

    score = (winrate*0.4) + (pf*0.4) + (expectancy*0.2)

    return max(0, min(score, 1))
